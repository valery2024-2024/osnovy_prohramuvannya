using System;
using System.IO;

class Program
{
    static void Main()
    {
        // Шлях до файлу з числами
        string path = @"D:\teaching_2025-2026\osnovy_prohramuvannya\lab1v1\tsili_chysla.txt";

        // Читання вмісту файлу
        string text = File.ReadAllText(path);

        // Перетворення рядка у масив чисел
        int[] numbers = Array.ConvertAll(
            text.Split(' ', StringSplitOptions.RemoveEmptyEntries),
            int.Parse
        );

        // Вивід масиву
        Console.WriteLine("Масив з файлу:");
        foreach (int n in numbers)
            Console.Write(n + " ");
        Console.WriteLine();

        // Мінімальне та максимальне
        int min = int.MaxValue, max = int.MinValue;
        foreach (int n in numbers)
        {
            if (n < min) min = n;
            if (n > max) max = n;
        }

        Console.WriteLine($"Мінімальне значення: {min}");
        Console.WriteLine($"Максимальне значення: {max}");
        Console.WriteLine($"Кількість елементів: {numbers.Length}");
    }
}
